TEMPLATE_FUNCTIONS:
// Meta title template
function getMetaTitleTemplate(primaryKeyword, secondaryKeywords, pageType, targetAudience) {
  let pageTypeSpecific = '';
  
  switch(pageType) {
    case 'homepage':
      pageTypeSpecific = `\n- Include the brand/company name\n- Focus on the overall value proposition rather than specific products/services\n- Consider a more evergreen approach as the homepage title rarely changes`;
      break;
    case 'product-page':
      pageTypeSpecific = `\n- Include product-specific details like model, size, or key feature\n- Consider including price point indicators (luxury, affordable, etc.)\n- Incorporate purchase intent language`;
      break;
    case 'category-page':
      pageTypeSpecific = `\n- Indicate the breadth of selection (e.g., "Wide Range", "Complete Collection")\n- Consider including category-specific qualifiers\n- Use language that helps users browse and compare`;
      break;
    case 'blog-post':
      pageTypeSpecific = `\n- Consider using numbers (e.g., "7 Ways to...")\n- Include year if content is time-sensitive\n- Use question formats if appropriate for the content`;
      break;
    case 'landing-page':
      pageTypeSpecific = `\n- Include strong call-to-action language\n- Emphasize unique selling proposition\n- Consider conversion-focused language`;
      break;
    default:
      pageTypeSpecific = '';
  }
  
  return `Create 10 SEO-optimized meta title options for a ${pageType} about "${primaryKeyword}" targeting ${targetAudience}. For each title:

1. Include the primary keyword "${primaryKeyword}" naturally
2. Incorporate at least one of these secondary keywords where appropriate: ${secondaryKeywords}
3. Keep the length between 50-60 characters to avoid truncation in search results
4. Include a compelling benefit or unique selling point
5. Use power words that encourage clicks
6. Maintain a brand voice that appeals to ${targetAudience}${pageTypeSpecific}

For each title option, provide:
- The exact character count
- An explanation of why it would be effective for both SEO and click-through rate
- Any potential SERP features it might help trigger

Rank the titles in order of recommended effectiveness, and explain your reasoning for the top choice.`;
}

// Meta description template
function getMetaDescriptionTemplate(primaryKeyword, secondaryKeywords, pageType, targetAudience) {
  let pageTypeSpecific = '';
  
  switch(pageType) {
    case 'homepage':
      pageTypeSpecific = `\n- Summarize the overall value proposition of the brand/company\n- Include mention of key product/service categories\n- Use language that establishes authority in the industry`;
      break;
    case 'product-page':
      pageTypeSpecific = `\n- Highlight key product features and benefits\n- Include pricing information or special offers if applicable\n- Use language that addresses purchase intent`;
      break;
    case 'category-page':
      pageTypeSpecific = `\n- Mention the range of products/items available in this category\n- Include category-specific qualifiers or filters\n- Use language that helps users understand the selection available`;
      break;
    case 'blog-post':
      pageTypeSpecific = `\n- Tease the key insights or takeaways from the post\n- Consider mentioning the format (guide, list, case study)\n- Use language that establishes expertise on the topic`;
      break;
    case 'landing-page':
      pageTypeSpecific = `\n- Include a clear, compelling call-to-action\n- Emphasize the unique value proposition\n- Address a specific pain point the landing page solves`;
      break;
    default:
      pageTypeSpecific = '';
  }
  
  return `Generate 5 SEO-optimized meta descriptions for a ${pageType} about "${primaryKeyword}" targeting ${targetAudience}. Each description should:

1. Include the primary keyword "${primaryKeyword}" naturally
2. Incorporate at least one secondary keyword from: ${secondaryKeywords}
3. Be between 140-155 characters to avoid truncation in search results
4. Include a clear value proposition for the reader
5. Contain a compelling call-to-action
6. Be written in a tone appropriate for ${targetAudience}${pageTypeSpecific}

For each description, provide:
- The exact character count
- Which secondary keywords were incorporated
- How it addresses user intent for this search
- Why it would encourage clicks from search results

Also provide guidance on which description would work best for different search intents (informational, commercial, transactional) related to this topic.`;
}

// Heading structure template
function getHeadingStructureTemplate(primaryKeyword, secondaryKeywords, pageType, targetAudience) {
  let pageTypeSpecific = '';
  
  switch(pageType) {
    case 'homepage':
      pageTypeSpecific = `\n- Focus on broad category headings that help with navigation\n- Include headings for key value propositions and USPs\n- Consider testimonial or social proof section headings`;
      break;
    case 'product-page':
      pageTypeSpecific = `\n- Include headings for product features, specifications, benefits\n- Consider headings for reviews, FAQs, and related products\n- Use headings to break up technical information into digestible sections`;
      break;
    case 'category-page':
      pageTypeSpecific = `\n- Include headings for different product subcategories or filters\n- Consider headings for featured or bestselling items\n- Use headings to highlight category-specific buying guides`;
      break;
    case 'blog-post':
      pageTypeSpecific = `\n- Structure headings to create a logical narrative flow\n- Include question-based headings for featured snippet opportunities\n- Consider using numbered headings for list-based content`;
      break;
    case 'landing-page':
      pageTypeSpecific = `\n- Include benefit-focused headings that emphasize value\n- Use headings to address potential objections\n- Consider headings for testimonials, guarantees, and calls-to-action`;
      break;
    default:
      pageTypeSpecific = '';
  }
  
  return `Create a comprehensive heading structure (H1-H4) for a ${pageType} about "${primaryKeyword}" targeting ${targetAudience}. The heading structure should:

1. Include an SEO-optimized H1 that incorporates the primary keyword "${primaryKeyword}"
2. Provide 5-7 H2 headings that cover the main subtopics and incorporate secondary keywords: ${secondaryKeywords}
3. For each H2, suggest 2-4 H3 subheadings that break down the content further
4. Where appropriate, include H4 headings for detailed sections
5. Ensure a logical content flow that guides the reader through the information
6. Incorporate question-based headings where appropriate for featured snippet opportunities${pageTypeSpecific}

For each heading:
- Explain its purpose in the overall content structure
- Identify which keywords it targets
- Note any featured snippet or SERP feature opportunities
- Provide a brief description of what content should go under this heading

Also include recommendations for the ideal word count for each major section to ensure comprehensive coverage of the topic.`;
}

// Image alt text template
function getImageAltTextTemplate(primaryKeyword, pageType, targetAudience) {
  let imageTypes = '';
  
  switch(pageType) {
    case 'homepage':
      imageTypes = `\n- Hero/banner image\n- Product/service category thumbnails\n- Team or company photos\n- Testimonial author images\n- Award or certification logos\n- Partner/client logos`;
      break;
    case 'product-page':
      imageTypes = `\n- Main product image\n- Product detail/feature close-ups\n- Product in use/context\n- Size/scale reference images\n- Color/style variants\n- Packaging images`;
      break;
    case 'category-page':
      imageTypes = `\n- Category banner image\n- Featured product thumbnails\n- Comparison charts/graphics\n- Category icon/symbol\n- Lifestyle images showing category products in use`;
      break;
    case 'blog-post':
      imageTypes = `\n- Featured/hero image\n- Step-by-step tutorial images\n- Infographics or data visualizations\n- Screenshots or examples\n- Author photo\n- Related concept illustrations`;
      break;
    case 'landing-page':
      imageTypes = `\n- Hero/banner image\n- Product/service showcase images\n- Benefit illustration graphics\n- Trust indicators (certifications, awards)\n- Customer testimonial photos\n- Call-to-action button/graphic`;
      break;
    default:
      imageTypes = '';
  }
  
  return `Generate SEO-optimized alt text for 8 images on a ${pageType} about "${primaryKeyword}" targeting ${targetAudience}. For this page, assume the following image types:${imageTypes}

For each image, create alt text that:
1. Naturally incorporates relevant keywords without keyword stuffing
2. Accurately describes what's shown in the image
3. Provides context relevant to the surrounding content
4. Highlights key selling points when relevant
5. Keeps length between 50-125 characters
6. Follows accessibility best practices

For each alt text suggestion, provide:
- The exact character count
- Which keywords it incorporates
- How it supports both SEO and accessibility goals
- Any specific scenarios where this image and alt text would be particularly effective

Also include general best practices for image optimization beyond alt text (file naming, compression, lazy loading, etc.) for this type of page.`;
}

// Internal linking template
function getInternalLinkingTemplate(primaryKeyword, pageType) {
  let pageTypeSpecific = '';
  
  switch(pageType) {
    case 'homepage':
      pageTypeSpecific = `\n\nSpecific to homepage internal linking:\n- Recommend primary navigation structure and categories\n- Suggest featured content sections that should link to key pages\n- Provide guidance on footer link structure`;
      break;
    case 'product-page':
      pageTypeSpecific = `\n\nSpecific to product page internal linking:\n- Suggest related/complementary product links\n- Recommend category and subcategory linking structure\n- Provide guidance on linking to buying guides or educational content`;
      break;
    case 'category-page':
      pageTypeSpecific = `\n\nSpecific to category page internal linking:\n- Recommend subcategory linking structure\n- Suggest featured product linking strategy\n- Provide guidance on breadcrumb implementation`;
      break;
    case 'blog-post':
      pageTypeSpecific = `\n\nSpecific to blog post internal linking:\n- Suggest related article links based on topic clusters\n- Recommend product/service page links where relevant\n- Provide guidance on sidebar and author bio linking`;
      break;
    case 'landing-page':
      pageTypeSpecific = `\n\nSpecific to landing page internal linking:\n- Recommend minimal but strategic internal links to avoid distracting from conversion\n- Suggest trust-building links (about, testimonials, case studies)\n- Provide guidance on post-conversion page linking`;
      break;
    default:
      pageTypeSpecific = '';
  }
  
  return `Develop a comprehensive internal linking strategy for a ${pageType} about "${primaryKeyword}". Assume the website already has the following content types:
- Product/service pages
- Category pages
- Blog posts/articles
- About/company pages
- FAQ/resource pages

For this ${pageType}, provide:

1. Strategic internal linking recommendations:
   - 10-12 specific anchor text suggestions for internal links to existing content
   - For each link, explain the contextual relevance and SEO benefit
   - Recommendations for natural link placement within the content flow
   - Suggestions for reciprocal linking from existing content to this page

2. Technical implementation guidance:
   - Optimal number of internal links for this page type
   - Anchor text variation recommendations (exact match, partial match, branded, etc.)
   - Link priority hierarchy (which links should receive more prominent placement)
   - How to avoid over-optimization penalties${pageTypeSpecific}

3. Content silo and topic cluster strategy:
   - How this page should fit within the overall site architecture
   - Topic clusters this page should belong to
   - Pillar content this page should link to
   - Supporting content that should link to this page

Include specific examples of anchor text phrasing that incorporates relevant keywords while maintaining natural language flow.`;
}

// Content optimization template
function getContentOptimizationTemplate(primaryKeyword, secondaryKeywords, pageType, targetAudience) {
  return `Create a comprehensive content optimization guide for a ${pageType} about "${primaryKeyword}" targeting ${targetAudience}. This guide should include:

1. Keyword optimization strategy:
   - How to naturally incorporate the primary keyword "${primaryKeyword}" (recommended density and placement)
   - How to integrate these secondary keywords: ${secondaryKeywords}
   - Semantic and LSI keyword suggestions to include
   - Terms and phrases to avoid (negative keywords)

2. Content structure recommendations:
   - Ideal content length based on competitive analysis
   - Paragraph length and structure for readability
   - Sentence structure variety and readability score targets
   - Subheading usage and keyword incorporation

3. Engagement optimization:
   - Hooks and opening paragraph techniques
   - Storytelling elements to incorporate
   - Transition phrases between sections
   - Conclusion and call-to-action recommendations

4. SEO enhancement elements:
   - List/table/bullet point usage for featured snippets
   - Question-and-answer sections for PAA opportunities
   - Multimedia integration recommendations
   - Schema markup suggestions

5. Content differentiation strategy:
   - Unique angles to explore on this topic
   - Data points and statistics to incorporate
   - Expert quote or insight opportunities
   - Originality factors to set this content apart

For each recommendation, explain both the SEO benefit and the user experience improvement it provides. Include specific examples of phrasing, formatting, and structure that would work well for this specific topic and audience.`;
}
